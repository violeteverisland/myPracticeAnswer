class Solution {
    public int scoreOfString(String s) {

        if(s == null || s.length()<2){
            return 0;
        }
        int score = 0;
        int charAscii = s.charAt(0);
        for(int i=0;i<s.length()-1;i++){
            score += Math.abs(s.charAt(i+1)-s.charAt(i));
        }
        return score;
    }
}
