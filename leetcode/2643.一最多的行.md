class Solution {
    public int[] rowAndMaximumOnes(int[][] mat) {
        int onesNum=0;
        int rowMax=0;
        for(int i=0;i<mat.length;i++){
            int onesCount=0;
            for(int j=0;j<mat[i].length;j++){
                if(mat[i][j] == 1){
                    onesCount ++;
                }
            }
            if(onesCount>onesNum){
                onesNum = onesCount;
                rowMax=i;  
            }
        }
        return new int[]{rowMax,onesNum};
    }
}
